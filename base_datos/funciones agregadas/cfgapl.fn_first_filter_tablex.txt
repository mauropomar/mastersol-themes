CREATE OR REPLACE FUNCTION cfgapl.fn_first_filter_tablex
(
  IN  table_name  varchar = ''::character varying,
  IN  id_rol      uuid    = NULL::uuid,
  IN  id_user     uuid    = NULL::uuid,
  IN  "start"     integer = 0,
  IN  "limit"     integer = 20
)
RETURNS jsonb AS
$$
declare
res_data jsonb = null;
name_table varchar = $1;
begin
    EXECUTE 'SELECT jsonb_agg(rowset.*) FROM(SELECT dat.* FROM '|| name_table ||' dat WHERE dat.nombre ilike ''%tarea1%'' OFFSET '|| $4 ||' LIMIT '|| $5 ||' )rowset ' INTO res_data;
    return res_data; 
end
$$
LANGUAGE 'plpgsql';

  OWNER TO postgres;
